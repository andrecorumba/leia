    '''
    Remove todos os arquivos tempor√°rios usados pela aplica√ß√£o.

    Parameters:
        folder_path (str): Caminho da pasta onde os arquivos tempor√°rios est√£o armazenados.

    Example:
        >>> clean_folder('/uploads')
    '''

        for filename in os.listdir(folder_path):
